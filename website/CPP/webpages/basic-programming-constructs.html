12. Write a C++ program to count the letters, spaces, numbers and other characters of an input string

Answer 12:

#include <iostream>
#include <string>

using namespace std;


int main(){

    string text;
    int letters = 0;
    int spaces = 0;
    int numbers = 0;
    int others = 0;

    cout << "Enter a string: ";
    getline(cin, text);

    for (int i = 0; i < text.length(); i++){
        char ch = text[i];

        if ((ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z')){
            letters++;
        }

        else if (ch == ' '){
            spaces++;
        }

        else if (ch >= '0' && ch <= '9'){
            numbers++;
        }

        else{
            others++;
        }
    }

    cout << "Letters: " << letters << "\n";
    cout << "Spaces: " << spaces << "\n";
    cout << "Numbers: " << numbers << "\n";
    cout << "Other characters: " << others ;
    return 0;
}
